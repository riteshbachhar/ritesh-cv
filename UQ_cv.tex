% ---- Research Experience ----
\section{Research Experience}

\begin{onecolentry}
  \textbf{Graduate Research Assistant } \hfill \textit{September 2021 - Present}\\
  \textit{Department of Physics, URI}
  
  \vspace{0.1 cm}

  \textit{Project 1: Uncertainty Quantification for Functional Data Surrogates (SEOBNRv4CE)}
  \begin{highlights}
    \item Developed a probabilistic surrogate model to quantify epistemic uncertainty in 
    \textbf{functional data} (gravitational waveforms), addressing systematic biases arising 
    from imperfect calibration against numerical relativity.
    \item Utilized \textbf{Gaussian Process Regression (GPR)} to model calibration error distributions 
    across high-dimensional parameter space, preventing overconfident parameter estimation.
    \item Implemented frequency-domain amplitude and phase deviation models using 
    \textbf{cubic splines} to capture non-smooth features, ensuring efficient and stable waveform evaluation.
    \item Conducted rigorous \textbf{Verification and Validation (V\&V)} of the UQ method by testing 
    against simulated injections in Advanced LIGO noise; demonstrated that uncertainty-aware analysis 
    reduced normalized biases by a factor of $\sim 1.5$ for spinning binary configurations.
    \item Integrated the model into the \texttt{bilby} Bayesian inference pipeline for production-ready 
    analysis and engineered a fully documented, pip-installable Python package.
    \item Links: \href{https://github.com/riteshbachhar/SEOBNRv4CE}{GitHub Repository} | \href{https://github.com/riteshbachhar/SEOBNRv4CE/blob/main/tutorial/SEOBNRv4CE.ipynb}{Tutorial Notebook}
  \end{highlights}

  \vspace{0.2 cm}

  \textit{Project 2: Scalable Surrogate Modeling via Dimensionality Reduction (BHPTNRSur)}
  \begin{highlights}
    \item Designed a high-fidelity surrogate model for binary black hole mergers, integrating 
    Perturbation Theory with Numerical Relativity to model complex \textbf{functional data} across 
    a wide parameter space ($q \in [3, 1000]$, $\chi_1 \in [-0.6, 0.8]$).
    \item Engineered a scalable training pipeline using \textbf{SVD-based dimensionality reduction} 
    to project high-dimensional waveform data into a compact latent space modeled by scalar 
    \textbf{Gaussian Process Regression (GPR)}.
    \item Achieved a $10^6\times$ speedup over numerical simulations with relative $L_2$ error $<10^{-5}$, 
    enabling the first fast evaluation of intermediate mass-ratio binaries for Bayesian inference.
    \item Built automated, parallelized data generation workflows on \textbf{HPC systems} (utilizing MPI/Slurm), 
    optimizing resource usage for large-scale model training across distributed clusters.
    \item Integrated the final model into the SXS Collaboration's production infrastructure and the 
    \texttt{gwsurrogate} package, delivering a robust, pip-installable tool currently used by multiple 
    external research groups.
    \item Links: \href{https://github.com/BlackHolePerturbationToolkit/BHPTNRSurrogate}{GitHub Repository} 
    | \href{https://github.com/BlackHolePerturbationToolkit/BHPTNRSurrogate/blob/main/tutorials/BHPTNRSur2dq1e3.ipynb}{Tutorial Notebook}
  \end{highlights}
\end{onecolentry}
